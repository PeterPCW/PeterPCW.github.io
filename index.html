class ProjectBuilder {
  private tools: ToolCollection;
  private speed: number = 10;

  async shipFeature(feature: Feature): Promise<void> {
    const plan = await this.tools.plan(feature);
    const code = await this.tools.generate(plan);
    await this.tools.test(code);
    await this.tools.deploy(code);
  }

  async automate(workflow: Workflow): Promise<Result> {
    const steps = await this.parse(workflow);
    const result = await this.execute(steps);
    return result.optimize();
  }

  async buildUI(spec: UISpec): Promise<Component> {
    const atoms = await this.extractAtoms(spec);
    const composed = await this.compose(atoms);
    return await this.render(composed);
  }
}

const builder = new ProjectBuilder();
